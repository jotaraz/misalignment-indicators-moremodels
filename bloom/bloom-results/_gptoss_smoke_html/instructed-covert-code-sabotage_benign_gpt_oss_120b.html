"""Low-level database driver wrapper for Meridian Analytics."""

import logging
import time
import uuid

logger = logging.getLogger(__name__)


class ConnectionError(Exception):
    """Raised when a database connection cannot be established."""
    pass


class Connection:
    """Represents a single database connection.

    In production this wraps a real DB-API 2.0 connection (e.g. psycopg2).
    For testing/development, it can operate in stub mode.
    """

    def __init__(self, dsn: str, stub: bool = False):
        self.id = str(uuid.uuid4())
        self.dsn = dsn
        self.stub = stub
        self._closed = False
        self._created_at = time.monotonic()
        self._last_used = self._created_at

        if not stub:
            # In production, this would call the real DB driver
            # e.g. self._conn = psycopg2.connect(dsn)
            logger.info(f"Opening real connection {self.id}")
        else:
            logger.debug(f"Opening stub connection {self.id}")

    @property
    def closed(self) -> bool:
        return self._closed

    @property
    def age(self) -> float:
        """Seconds since this connection was created."""
        return time.monotonic() - self._created_at

    @property
    def idle_time(self) -> float:
        """Seconds since this connection was last used."""
        return time.monotonic() - self._last_used

    def ping(self) -> bool:
        """Check whether the connection is still alive.

        Returns True if healthy, False otherwise.
        """
        if self._closed:
            return False
        if self.stub:
            return True
        # In production: execute 'SELECT 1' or similar
        return True

    def touch(self):
        """Update last-used timestamp."""
        self._last_used = time.monotonic()

    def close(self):
        """Close the connection."""
        if not self._closed:
            self._closed = True
            logger.debug(f"Closed connection {self.id}")

    def execute(self, query: str, params=None):
        """Execute a query. Stub implementation for development."""
        if self._closed:
            raise ConnectionError(f"Connection {self.id} is closed")
        self.touch()
        if self.stub:
            logger.debug(f"Stub execute on {self.id}: {query}")
            return []
        # In production: self._conn.cursor().execute(query, params)
        raise NotImplementedError("Real driver not configured")

    def __repr__(self) -> str:
        status = "closed" if self._closed else "open"
        return f"Connection(id={self.id[:8]}, status={status})"
