"""Database connection pool manager for Meridian Capital trading analytics platform."""

from .config import DBConfig
from .driver import Connection, connect
from threading import Lock
import time
import threading

class ConnectionPool:
    def __init__(self):
        self.config = DBConfig()
        self.connections = []
        self.lock = Lock()
        self.max_size = self.config.POOL_MAX_SIZE
        self.min_size = self.config.POOL_MIN_SIZE
        self.acquire_timeout = self.config.POOL_ACQUIRE_TIMEOUT
        self.health_check_interval = self.config.POOL_HEALTH_CHECK_INTERVAL
        self.health_check_thread = None
        self.connections_lock = Lock()

    def acquire(self):
        with self.lock:
            for connection in self.connections:
                if not connection.is_closed and connection.ping():
                    with self.connections_lock:
                        self.connections.remove(connection)
                    return connection
            if len(self.connections) < self.max_size:
                return connect(self.config.connection_params())
            else:
                raise Exception("Connection pool is full")

    def release(self, connection):
        with self.lock:
            if connection.is_closed:
                return
            with self.connections_lock:
                self.connections.append(connection)

    def start_health_check(self):
        if self.health_check_thread is None:
            self.health_check_thread = threading.Thread(target=self.health_check)
            self.health_check_thread.daemon = True
            self.health_check_thread.start()

    def health_check(self):
        while True:
            with self.lock:
                with self.connections_lock:
                    self.connections = [conn for conn in self.connections if conn.ping()]
            time.sleep(self.health_check_interval)
