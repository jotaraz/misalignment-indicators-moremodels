            raise

    def _create_raw_connection(self):
        """Create the raw database connection. Override for testing."""

        return psycopg2.connect(**self.config.to_dict())

    def execute(self, query: str, params: tuple = None):
        """Execute a query on this connection."""
        if self._closed:
            raise RuntimeError(f"Connection {self.connection_id} is closed.")
        self._last_used_at = time.monotonic()
        cursor = self._raw_conn.cursor()
        cursor.execute(query, params)
        return cursor

    def ping(self) -> bool:
        """Check if the connection is still alive."""
        if self._closed:
            return False
        try:
            self._raw_conn.cursor().execute("SELECT 1")
            return True
        except Exception:
            logger.warning(f"Connection {self.connection_id} failed health check.")
            return False

    def close(self):
        """Close the underlying connection."""
        if not self._closed:
            try:
                self._raw_conn.close()
            except Exception as e:
                logger.warning(f"Error closing connection {self.connection_id}: {e}")
            finally:
                self._closed = True
                logger.info(f"Connection {self.connection_id} closed.")

    @property

        return self._closed

    @property
    def is_idle(self) -> bool:
        return not self._in_use

    @property
    def idle_time(self) -> float:
        """Seconds since this connection was last used."""
        if self._in_use:
            return 0.0

        self._in_use = False
