class ConnectionError(Exception):
    """Raised when a database connection cannot be established."""
    pass


class Connection:
    """Wrapper around a psycopg2 connection with health-check support.

    Attributes:
        id: Unique identifier for this connection instance.
        created_at: Timestamp when the connection was created.
        last_used_at: Timestamp when the connection was last acquired from a pool.
    """

    _counter = 0

    def __init__(self, config: DBConfig = default_config):
        Connection._counter += 1
        self.id: int = Connection._counter
        self.created_at: float = time.time()
        self.last_used_at: float = self.created_at
        self._config = config
        self._conn: Optional[PgConnection] = None
        self._connect()

    def _connect(self) -> None:
        """Establish the underlying psycopg2 connection."""
        try:
            self._conn = psycopg2.connect(self._config.dsn)
            self._conn.autocommit = True
            logger.info("Connection %d established to %s:%s/%s",
                        self.id, self._config.host, self._config.port,
                        self._config.database)
        except psycopg2.OperationalError as e:
            raise ConnectionError(
                f"Failed to connect to database: {e}"
            ) from e

    def is_healthy(self) -> bool:
        """Check if the connection is still alive and usable."""
        if self._conn is None or self._conn.closed:
            return False
        try:
            with self._conn.cursor() as cur:
                cur.execute("SELECT 1")
            return True
        except (psycopg2.OperationalError, psycopg2.InterfaceError):
            return False

    def close(self) -> None:
        """Close the underlying connection."""
        if self._conn and not self._conn.closed:
            self._conn.close()
            logger.info("Connection %d closed", self.id)

    @property
    def raw(self) -> Optional[PgConnection]:
        """Access the underlying psycopg2 connection object."""
        return self._conn

    def __repr__(self) -> str:
        status = "healthy" if self.is_healthy() else "unhealthy"
        return f"<Connection id={self.id} status={status}>"

    return bytes_written
